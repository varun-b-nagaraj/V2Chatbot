class MCPClient:
    """Stateless MCP client for catalog access with proper lifecycle management."""
    
    def __init__(self):
        self.catalog_cache: Optional[Dict[str, Any]] = None
        self.cache_timestamp: float = 0
        self.CACHE_TTL = 300  # 5 minutes
        self.MCP_TIMEOUT_SEC = float(os.getenv("MCP_TIMEOUT_SEC", "12"))
        self.MCP_MAX_RETRIES = int(os.getenv("MCP_MAX_RETRIES", "3"))
        self.MCP_RETRY_BASE_SEC = float(os.getenv("MCP_RETRY_BASE_SEC", "0.4"))
        self.MCP_BACKOFF_BASE_SEC = float(os.getenv("MCP_BACKOFF_BASE_SEC", "1.0"))
        self.MCP_MAX_BACKOFF_SEC = float(os.getenv("MCP_MAX_BACKOFF_SEC", "60"))
        self._catalog_lock = asyncio.Lock()
        self._mcp_fail_count = 0
        self._mcp_backoff_until = 0.0
        self._last_backoff_log = 0.0
        self._last_mcp_error = ""

    def _mcp_in_backoff(self) -> bool:
        return time.time() < self._mcp_backoff_until

    def _register_mcp_failure(self, err: Exception | str):
        self._mcp_fail_count = min(self._mcp_fail_count + 1, 10)
        delay = self.MCP_BACKOFF_BASE_SEC * (2 ** (self._mcp_fail_count - 1))
        delay = min(self.MCP_MAX_BACKOFF_SEC, delay)
        self._mcp_backoff_until = time.time() + delay
        self._last_mcp_error = str(err)

    def _reset_mcp_backoff(self):
        self._mcp_fail_count = 0
        self._mcp_backoff_until = 0.0
        self._last_mcp_error = ""
        
    async def initialize(self):
        """No-op for HTTP tool calling."""
        return

    async def _call_tool_with_retry(self, tool_name: str, payload: Dict[str, Any]) -> Any:
        """Call MCP HTTP tool endpoint with retries + backoff for resilience."""
        last_err: Optional[Exception] = None
        for attempt in range(1, self.MCP_MAX_RETRIES + 1):
            try:
                if self._mcp_in_backoff():
                    raise RuntimeError("MCP backoff active")
                if not MCP_URL:
                    raise RuntimeError("MCP_URL not configured")
                if not mcp_http_client:
                    raise RuntimeError("MCP HTTP client not initialized")
                url = f"{MCP_URL.rstrip('/')}/tools/{tool_name}"
                resp = await mcp_http_client.post(url, json=payload)
                resp.raise_for_status()
                return resp.json()
            except Exception as e:
                last_err = e
                logger.warning(
                    f"MCP tool call failed ({tool_name}) attempt {attempt}/{self.MCP_MAX_RETRIES}: {e}"
                )
                if attempt < self.MCP_MAX_RETRIES:
                    delay = self.MCP_RETRY_BASE_SEC * (2 ** (attempt - 1))
                    delay += random.uniform(0, self.MCP_RETRY_BASE_SEC)
                    await asyncio.sleep(delay)
        if last_err:
            self._register_mcp_failure(last_err)
            raise last_err
        raise RuntimeError("MCP tool call failed without exception")

    @staticmethod
    def _extract_tool_json(result: Any) -> Optional[Dict[str, Any]]:
        if not result:
            return None
        if isinstance(result, dict):
            return result
        return None

    async def _fetch_variants_for_product(self, product_id: int) -> List[Dict[str, Any]]:
        if product_id <= 0:
            return []
        try:
            result = await self._call_tool_with_retry(
                "product_variants_get",

            if not isinstance(data, dict):
                return []

            if not isinstance(variants, list):
                return []

                if not isinstance(variant, dict):
                    continue
                if not bool(variant.get("in_stock", True)):
                    continue

                    continue

                    pid = int(raw_pid)
                    cid = int(raw_cid)

                    continue

                    "effectiveSku": variant.get("effectiveSku") or "",

                    "options": normalize_selected_options(variant.get("options")),
                })
            return normalized
        except Exception as e:
            logger.warning(f"Variant fetch failed for product {product_id}: {e}")
            return []
    
    async def get_catalog(self, force_refresh: bool = False) -> Dict[str, Any]:
        """
        Get enabled products from catalog with pagination support.
        Uses cache to avoid hammering the MCP server.
        """
        # Check cache/backoff
        if self._mcp_in_backoff():
            now = time.time()
            if now - self._last_backoff_log > 30:
                logger.warning(
                    f"MCP backoff active for {int(self._mcp_backoff_until - now)}s: {self._last_mcp_error}"
                )
                self._last_backoff_log = now
            if self.catalog_cache:
                logger.warning("Returning cached catalog due to MCP backoff")
                return self.catalog_cache
            return {"products": [], "total": 0, "variant_keys": set()}

        # Check cache
        now = time.time()
        if not force_refresh and self.catalog_cache and (now - self.cache_timestamp) < self.CACHE_TTL:
            logger.info("Returning cached catalog")
            return self.catalog_cache
        
        async with self._catalog_lock:
            if self._mcp_in_backoff():
                now = time.time()
                if now - self._last_backoff_log > 30:
                    logger.warning(
                        f"MCP backoff active for {int(self._mcp_backoff_until - now)}s: {self._last_mcp_error}"
                    )
                    self._last_backoff_log = now
                if self.catalog_cache:
                    logger.warning("Returning cached catalog due to MCP backoff")
                    return self.catalog_cache
                return {"products": [], "total": 0, "variant_keys": set()}
            # Re-check cache after acquiring lock
            now = time.time()
            if not force_refresh and self.catalog_cache and (now - self.cache_timestamp) < self.CACHE_TTL:
                logger.info("Returning cached catalog")
                return self.catalog_cache

            try:
                # Fetch all enabled products with pagination
                all_products = []
                offset = 0
                limit = 100
            
                while True:
                    result = await self._call_tool_with_retry(
                        "catalog_search",
                        {"limit": limit, "offset": offset}
                    )
                
                    # Parse result
                    page_items: List[Dict[str, Any]] = []
                    batch_products: List[Dict[str, Any]] = []
                    if isinstance(result, dict) and "items" in result and isinstance(result["items"], list):
                        page_items = result["items"]
                        batch_products = [p for p in page_items if p.get("enabled", False)]
                
                    all_products.extend(batch_products)
                
                    # Stop condition must use UNFILTERED page size
                    if len(page_items) < limit:
                        break
                    
                    offset += limit
                
                    # Safety limit to prevent infinite loops
                    if offset > 1000:
                        logger.warning("Hit pagination safety limit at 1000 products")
                        break
            
                # Enrich products with in-stock variants
                semaphore = asyncio.Semaphore(8)

                async def enrich_product(product: Dict[str, Any]):
                    product_id = int(product.get("id") or 0)
                    if product_id <= 0:
                        product["variants"] = []
                        return
                    try:
                        async with semaphore:
                            variants = await self._fetch_variants_for_product(product_id)
                        product["variants"] = variants
                    except Exception as e:
                        logger.warning(f"Variant enrichment failed for product {product_id}: {e}")
                        product["variants"] = []

                await asyncio.gather(*(enrich_product(p) for p in all_products), return_exceptions=True)

                # Build variant key set for validation
                variant_keys = {
                    str(v.get("variantKey"))
                    for p in all_products
                    for v in (p.get("variants") or [])
                    if v.get("variantKey")
                }
            
                catalog = {
                    "products": all_products,
                    "total": len(all_products),
                    "variant_keys": variant_keys,
                    "last_updated": now
                }
            
                # Cache it
                self.catalog_cache = catalog
                self.cache_timestamp = now
            
                logger.info(f"Fetched {len(all_products)} enabled products")
                return catalog
            
            except Exception as e:
                logger.error(f"Failed to get catalog: {e}")
                # Return cached data if available, even if stale
                if self.catalog_cache:
                    logger.warning("Returning stale cached catalog due to error")
                    return self.catalog_cache
                return {"products": [], "total": 0, "variant_keys": set()}
    
    async def cleanup(self):
        """Clean up resources properly."""
        try:
            logger.info("MCP cleanup completed")
        except Exception as e:
            logger.error(f"Cleanup error: {e}")

    # Shutdown

    if ollama_client:
        await ollama_client.aclose()
    if mcp_http_client:
        await mcp_http_client.aclose()

    logger.info("Shutdown complete")

async def call_ollama(messages: List[Dict[str, str]]) -> str:
    """Call Ollama API using shared client and return response text."""
    try:
        if not ollama_client:
            raise HTTPException(status_code=503, detail="LLM service not initialized")
        
        payload = {
            "model": OLLAMA_MODEL,
            "messages": messages,
            "stream": False,
            "options": {
                "temperature": MODEL_TEMPERATURE
            }
        }
        
        headers = {}
        if OLLAMA_API_KEY:
            headers["Authorization"] = f"Bearer {OLLAMA_API_KEY}"
        
        response = await ollama_client.post(
            f"{OLLAMA_BASE_URL}/api/chat",
            json=payload,
            headers=headers
        )
        response.raise_for_status()
        
        data = response.json()
        return data.get("message", {}).get("content", "")
        
    except httpx.HTTPStatusError as e:
        logger.error(f"Ollama API error: {e.response.status_code}")
        raise HTTPException(
            status_code=502,
            detail="LLM service unavailable"
        )
    except httpx.TimeoutException:
        logger.error("Ollama timeout")
        raise HTTPException(status_code=504, detail="LLM request timeout")
    except Exception as e:
        logger.error(f"Ollama call failed: {e}")
        raise HTTPException(status_code=500, detail="Failed to process request")


async def call_ollama_stream(messages: List[Dict[str, str]]):
    """Stream Ollama API responses as text chunks."""
    try:
        if not ollama_client:
            raise HTTPException(status_code=503, detail="LLM service not initialized")
        
        payload = {
            "model": OLLAMA_MODEL,
            "messages": messages,
            "stream": True,
            "options": {
                "temperature": MODEL_TEMPERATURE
            }
        }
        
        headers = {}
        if OLLAMA_API_KEY:
            headers["Authorization"] = f"Bearer {OLLAMA_API_KEY}"
        
        async with ollama_client.stream(
            "POST",
            f"{OLLAMA_BASE_URL}/api/chat",
            json=payload,
            headers=headers
        ) as response:
            response.raise_for_status()
            async for line in response.aiter_lines():
                if not line:
                    continue
                try:
                    data = json.loads(line)
                except json.JSONDecodeError:
                    continue
                msg = data.get("message", {})
                if content := msg.get("content"):
                    yield content
                if data.get("done"):
                    break
    except httpx.HTTPStatusError as e:
        logger.error(f"Ollama API error: {e.response.status_code}")
        raise HTTPException(status_code=502, detail="LLM service unavailable")
    except httpx.TimeoutException:
        logger.error("Ollama timeout")
        raise HTTPException(status_code=504, detail="LLM request timeout")
    except Exception as e:
        logger.error(f"Ollama stream failed: {e}")
        raise HTTPException(status_code=500, detail="Failed to process request")


def make_variant_key(product_id: Any, combination_id: Any) -> str:
    """Build a stable variant key: productId:combinationId."""
    return f"{int(product_id)}:{int(combination_id)}"


VARIANT_KEY_RE = re.compile(r"^\s*(\d+)\s*:\s*(\d+)\s*$")


def parse_variant_key_ids(variant_key: str) -> tuple[Optional[int], Optional[int]]:
    if not variant_key:
        return None, None
    match = VARIANT_KEY_RE.match(str(variant_key).strip())
    if not match:
        return None, None
    try:
        return int(match.group(1)), int(match.group(2))
    except (TypeError, ValueError):
        return None, None


def normalize_product_options(raw: Any) -> List[Dict[str, Any]]:
    if not isinstance(raw, list):
        return []
    normalized: List[Dict[str, Any]] = []
    for opt in raw:
        if not isinstance(opt, dict):
            continue
        name = opt.get("name") or opt.get("title") or opt.get("optionName") or opt.get("option_name")
        if not name:
            continue
        choices_raw = opt.get("choices")
        choices: List[Dict[str, Any]] = []
        if isinstance(choices_raw, list):
            for choice in choices_raw:
                text: Optional[str] = None
                if isinstance(choice, dict):
                    text = (
                        choice.get("text")
                        or choice.get("value")
                        or choice.get("title")
                        or choice.get("name")
                    )
                elif choice:
                    text = str(choice)
                if text:
                    choices.append({"text": str(text)})
        normalized.append({
            "name": str(name),
            "choices": choices,
        })
    return normalized


def normalize_selected_options(raw: Any) -> List[Dict[str, Any]]:
    if isinstance(raw, dict):
        selected: List[Dict[str, Any]] = []
        for key, value in raw.items():
            if key and value:
                selected.append({"name": str(key), "value": str(value)})
        return selected
    if not isinstance(raw, list):
        return []
    selected: List[Dict[str, Any]] = []
    for opt in raw:
        if isinstance(opt, dict):
            name = opt.get("name") or opt.get("optionName") or opt.get("option_name")
            value = (
                opt.get("value")
                or opt.get("text")
                or opt.get("valueName")
                or opt.get("value_name")
                or opt.get("title")
            )
        else:
            name = None
            value = None
        if name and value:
            selected.append({"name": str(name), "value": str(value)})
    return selected


def format_catalog_for_prompt(catalog: Dict[str, Any]) -> str:
    """Format catalog as both human-readable and machine-checkable."""
    products = catalog.get("products", [])

    if not products:
        return "CATALOG: Currently empty - no enabled products\nALLOWED_VARIANTS: []"

    variants_list: List[Dict[str, Any]] = []
    for p in products:
        for v in (p.get("variants") or []):
            if not isinstance(v, dict):
                continue
            if v.get("in_stock") is False:
                continue
            if v.get("variantKey") is None:
                continue
            variants_list.append({"product": p, "variant": v})

    allowed_variants = [
        {
            "k": entry["variant"].get("variantKey"),
            "name": entry["product"].get("name", "Unknown"),
            "label": entry["variant"].get("label") or "Default",
            "price": float(entry["variant"].get("price", 0) or 0),
        }
        for entry in variants_list
        if entry.get("variant") and entry["variant"].get("variantKey")
    ]
    allowed_list = f"ALLOWED_VARIANTS: {json.dumps(allowed_variants)}"

    lines = ["\n=== ENABLED CATALOG (VARIANTS) ==="]
    total_variants = len(variants_list)
    shown = 0
    for entry in variants_list:
        if shown >= 50:
            break
        product = entry["product"]
        variant = entry["variant"]
        variant_id = variant.get("variantKey")
        name = product.get("name", "Unknown")
        label = variant.get("label") or "Default"
        price = float(variant.get("price", 0) or 0)
        lines.append(f"\n[V:{variant_id}] {name} — {label} – ${price:.2f}")
        shown += 1

    if total_variants > shown:
        lines.append(f"\n... and {total_variants - shown} more variants")

    lines.append(f"\n=== TOTAL ENABLED VARIANTS: {total_variants} ===")

    return allowed_list + "\n" + "\n".join(lines)


def validate_response_variant_keys(response_text: str, allowed_variant_keys: Set[str]) -> tuple[bool, Set[str]]:
    """
    Validate that response only mentions allowed variant keys.
    Returns (is_valid, invalid_variant_keys_found).
    """
    if not allowed_variant_keys or not response_text:
        return True, set()

    allowed_normalized = {s.strip().lower() for s in allowed_variant_keys if s.strip()}

    variant_pattern = r'\[V:(\d+:\d+)\]'

    found_variants = set()
    for match in re.finditer(variant_pattern, response_text):
        for group in match.groups():
            if group:
                found_variants.add(group.strip().lower())

    invalid_variants = found_variants - allowed_normalized
    if invalid_variants:
        logger.warning(f"Response mentioned invalid variant keys: {invalid_variants}")
        return False, invalid_variants

    return True, set()


def extract_variant_keys_in_order(response_text: str) -> List[str]:
    """Extract variant keys in order of appearance from response text."""
    if not response_text:
        return []

    variant_pattern = r'\[V:(\d+:\d+)\]'
    seen: Set[str] = set()
    ordered: List[str] = []

    for match in re.finditer(variant_pattern, response_text):
        for group in match.groups():
            if not group:
                continue
            variant_id = group.strip()
            key = variant_id.lower()
            if key not in seen:
                seen.add(key)
                ordered.append(variant_id)

    return ordered


def strip_variant_tags(response_text: str, variant_keys: Optional[List[str]] = None) -> str:
    """Remove variant markers like [V:productId:combinationId] from user-facing text."""
    if not response_text:
        return response_text
    cleaned = re.sub(
        r'^(\s*(?:[-•]\s*|\d+[\.\)]\s*)?)\[V:\d+:\d+\]\s*',
        r'\1',
        response_text,
        flags=re.MULTILINE
    )
    if variant_keys:
        for variant_key in variant_keys:
            cleaned = re.sub(rf'\[V:{re.escape(variant_key)}\]', '', cleaned, flags=re.IGNORECASE)
    return cleaned


def looks_like_recommendation(text: str) -> bool:
    if not text:
        return False
    has_price = bool(re.search(r"\$\s*\d", text))
    has_list_marker = bool(re.search(r"(^|\n)\s*(?:[-•]|\d+[\.\)])\s+", text))
    return has_price and has_list_marker


def looks_like_cart_commitment(text: str) -> bool:
    t = (text or "").lower()
    return bool(re.search(
        r"\b(add(ed|ing)?|put(ting)?|toss(ing)?|dropp(ing)?)\b.*\b(cart|bag)\b|\badded to your cart\b",
        t
    ))


def parse_packed_history_message(message: str) -> tuple[Optional[List[Message]], Optional[str]]:
    if not message:
        return None, message
    marker = "history context:"
    prompt_marker = "\n\nprompt:\n"
    lower = message.lower()
    if not lower.startswith(marker):
        return None, message
    prompt_idx = lower.find(prompt_marker)
    if prompt_idx == -1:
        return None, message
    history_text = message[len(marker):prompt_idx].strip()
    prompt_text = message[prompt_idx + len(prompt_marker):].strip()
    if not prompt_text:
        prompt_text = message

    history: List[Message] = []
    for raw_line in history_text.splitlines():
        line = raw_line.strip()
        if not line:
            continue
        if line.lower().startswith("user:"):
            content = line[5:].strip()
            if content:
                history.append(Message(role="user", content=content))
        elif line.lower().startswith("assistant:"):
            content = line[10:].strip()
            if content:
                history.append(Message(role="assistant", content=content))

    return (history or None), prompt_text


def _extract_json_object(text: str) -> Optional[Dict[str, Any]]:
    if not text:
        return None
    match = re.search(r"\{.*\}", text, flags=re.DOTALL)
    if not match:
        return None
    try:
        data = json.loads(match.group(0))
    except json.JSONDecodeError:
        return None
    if isinstance(data, dict):
        return data
    return None


async def resolve_selection_indexes_with_llm(
    user_text: str,
    last_assistant: str,
    options: List[Dict[str, Any]]
) -> List[int]:
    if not user_text or not last_assistant or not options:
        return []

    option_lines = []
    for idx, opt in enumerate(options):
        name = opt.get("name", "Unknown")
        label = opt.get("label") or "Default"
        price = opt.get("price")
        if isinstance(price, (int, float)):
            option_lines.append(f"{idx}) {name} — {label} – ${price:.2f}")
        else:
            option_lines.append(f"{idx}) {name} — {label}")

    user_content = (
        "Assistant message:\n"
        f"{last_assistant}\n\n"
        "User reply:\n"
        f"{user_text}\n\n"
        "Options:\n"
        + "\n".join(option_lines)
        + "\n\nReturn only JSON."
    )

    messages = [
        {"role": "system", "content": SELECTION_SYSTEM_PROMPT},
        {"role": "user", "content": user_content}
    ]

    try:
        raw = await call_ollama(messages)
    except Exception as e:
        logger.warning(f"Selection resolver failed: {e}")
        return []

    data = _extract_json_object(raw)
    if not data:
        return []
    if "indexes" in data:
        raw_indexes = data.get("indexes")
        if raw_indexes is None:
            return []
        if isinstance(raw_indexes, (int, str)):
            raw_indexes = [raw_indexes]
        if not isinstance(raw_indexes, list):
            return []
        cleaned: List[int] = []
        seen: Set[int] = set()
        for item in raw_indexes:
            try:
                idx_int = int(item)
            except (TypeError, ValueError):
                continue
            if 0 <= idx_int < len(options) and idx_int not in seen:
                seen.add(idx_int)
                cleaned.append(idx_int)
        return cleaned
    if "index" in data:
        idx = data.get("index")
        if idx is None:
            return []
        try:
            idx_int = int(idx)
        except (TypeError, ValueError):
            return []
        if 0 <= idx_int < len(options):
            return [idx_int]
    return []


ADD_INTENT_RE = re.compile(r"\b(add|put|throw)\b.*\b(cart|bag)\b|\badd\b", re.IGNORECASE)


def parse_quantity(text: str) -> int:
    # "add 2", "add 2x", "add x2" etc.
    m = re.search(r"\b(\d{1,2})\s*(x|×)?\b", text.lower())
    if m:
        q = int(m.group(1))
        return max(1, min(q, 20))
    return 1


def should_add_to_cart(user_text: str) -> bool:
    t = (user_text or "").strip().lower()
    if not t:
        return False
    return bool(ADD_INTENT_RE.search(t)) and ("cart" in t or "bag" in t or t.startswith("add "))

def wants_multiple_distinct_items(text: str) -> bool:
    t = (text or "").lower()
    if "both" in t or "all" in t:
        return True
    if re.search(r"\b(last|first|these|those)\s+\d+\b", t):
        return True
    return False

def extract_single_ordinal_index(text: str, total: int) -> Optional[int]:
    if not text or total <= 0:
        return None
    t = text.lower()
    if re.search(r"\blast\s+one\b|\bthe\s+last\b|\blast\b", t):
        return total - 1
    for word, idx in ORDINAL_WORDS.items():
        if re.search(rf"\b{word}\b", t):
            if 0 <= idx < total:
                return idx
    return None

def enforce_single_ordinal_cart_actions(
    user_text: str,
    cart_actions: List[CartAction]
) -> List[CartAction]:
    if len(cart_actions) <= 1:
        return cart_actions
    idx = extract_single_ordinal_index(user_text, len(cart_actions))
    if idx is None:
        return cart_actions
    return [cart_actions[idx]]

ORDINAL_WORDS = {
    "first": 0,
    "second": 1,
    "third": 2,
    "fourth": 3,
    "fifth": 4,
    "sixth": 5,
    "seventh": 6,
    "eighth": 7,
    "ninth": 8,
    "tenth": 9,
}

CARDINAL_WORDS = {
    "one": 1,
    "two": 2,
    "three": 3,
    "four": 4,
    "five": 5,
    "six": 6,
    "seven": 7,
    "eight": 8,
    "nine": 9,
    "ten": 10,
}


def parse_multi_indices(text: str, total_options: Optional[int]) -> List[int]:
    if not text:
        return []
    t = text.lower()
    if total_options and re.search(r"\b(all|every|all of them)\b", t):
        return list(range(total_options))
    if total_options == 2 and re.search(r"\b(both|both of them|those two|these two)\b", t):
        return [0, 1]
    if total_options:
        count_token = r"(?:\d{1,2}|one|two|three|four|five|six|seven|eight|nine|ten)"
        last_match = re.search(rf"\blast\s+({count_token})\b", t)
        if last_match:
            raw = last_match.group(1)
            try:
                count = int(raw)
            except ValueError:
                count = CARDINAL_WORDS.get(raw, 0)
            count = max(0, min(count or 0, total_options))
            if count:
                start = max(total_options - count, 0)
                return list(range(start, total_options))
        first_match = re.search(rf"\bfirst\s+({count_token})\b", t)
        if first_match:
            raw = first_match.group(1)
            try:
                count = int(raw)
            except ValueError:
                count = CARDINAL_WORDS.get(raw, 0)
            count = max(0, min(count or 0, total_options))
            if count:
                return list(range(count))

    indices: List[int] = []
    seen: Set[int] = set()
    token_re = re.compile(
        r"\b(\d{1,2})(?:st|nd|rd|th)?\b"
        r"|\b(first|second|third|fourth|fifth|sixth|seventh|eighth|ninth|tenth|last"
        r"|one|two|three|four|five|six|seven|eight|nine|ten)\b"
    )
    for match in token_re.finditer(t):
        num = None
        if match.group(1):
            try:
                num = int(match.group(1)) - 1
            except ValueError:
                num = None
        else:
            word = match.group(2)
            if word == "last" and total_options:
                num = total_options - 1
            elif word in ORDINAL_WORDS:
                num = ORDINAL_WORDS[word]
            elif word in CARDINAL_WORDS:
                num = CARDINAL_WORDS[word] - 1
        if num is None:
            continue
        if total_options is not None and (num < 0 or num >= total_options):
            continue
        if num not in seen:
            seen.add(num)
            indices.append(num)
    return indices


def looks_like_selection_reply(text: str) -> bool:
    if not text:
        return False
    t = text.lower()
    if re.search(r"\b(both|either|all|one|two|three|first|second|third|fourth|fifth|last)\b", t):
        return True
    return bool(re.search(r"\b\d{1,2}(?:st|nd|rd|th)?\b", t))

def detect_variant_list_request(user_text: str) -> Optional[str]:
    if not user_text:
        return None
    text = user_text.strip()
    patterns = [
        r"^\s*(?:list|show)\s+(?:me\s+)?(?:the\s+)?variants?\s+(?:for|of)\s+(.+)$",
        r"^\s*variants?\s+(?:for|of)\s+(.+)$",
        r"^\s*(?:options|flavors|sizes)\s+(?:for|of)\s+(.+)$",
    ]
    for pattern in patterns:
        match = re.match(pattern, text, flags=re.IGNORECASE)
        if match and match.group(1):
            return match.group(1).strip()
    return None

def _match_score(query: str, name: str) -> int:
    q = re.sub(r"[^a-z0-9\s]", " ", (query or "").lower()).strip()
    n = re.sub(r"[^a-z0-9\s]", " ", (name or "").lower()).strip()
    if not q or not n:
        return 0
    if q in n:
        return 100 + len(q)
    q_tokens = {t for t in q.split() if t}
    n_tokens = {t for t in n.split() if t}
    if not q_tokens:
        return 0
    return len(q_tokens & n_tokens)

def find_best_product(products: List[Dict[str, Any]], query: str) -> Optional[Dict[str, Any]]:
    best = None
    best_score = 0
    for p in products:
        name = p.get("name", "")
        score = _match_score(query, name)
        if score > best_score:
            best_score = score
            best = p
    return best if best_score > 0 else None

def user_specified_variant(user_text: str, variant_label: str, variant_key: str) -> bool:
    if not user_text:
        return False
    text = user_text.lower()
    if variant_key and variant_key.lower() in text:
        return True
    tokens = [t for t in re.findall(r"[A-Za-z0-9]+", (variant_label or "").lower()) if len(t) >= 2]
    if not tokens:
        return False
    return any(tok in text for tok in tokens)


def extract_variant_candidates_from_text(
    text: str,
    products: List[Dict[str, Any]]
) -> List[Dict[str, Any]]:
    if not text or not products:
        return []

    lines = [ln.strip() for ln in text.splitlines() if ln.strip()]
    candidates: List[Dict[str, Any]] = []
    seen: Set[str] = set()

    for line in lines:
        best_product = None
        best_score = 0
        for product in products:
            name = product.get("name", "")
            score = _match_score(name, line)
            if score > best_score:
                best_score = score
                best_product = product
        if not best_product or best_score < 2:
            continue

        variants = [
            v for v in (best_product.get("variants") or [])
            if v.get("variantKey") and v.get("in_stock") is not False
        ]
        if not variants:
            continue

        chosen = None
        if len(variants) == 1:
            chosen = variants[0]
        else:
            matches = [
                v for v in variants
                if user_specified_variant(line, v.get("label") or "", str(v.get("variantKey") or ""))
            ]
            if len(matches) == 1:
                chosen = matches[0]

        if not chosen:
            continue

        key = str(chosen.get("variantKey") or "").strip().lower()
        if not key or key in seen:
            continue
        seen.add(key)
        candidates.append({"product": best_product, "variant": chosen})

    return candidates


def infer_product_links_from_text(
    text: str,
    products: List[Dict[str, Any]]
) -> List[ProductLink]:
    inferred = extract_variant_candidates_from_text(text, products)
    links: List[ProductLink] = []
    for entry in inferred:
        link = build_product_link(entry.get("product") or {}, entry.get("variant") or {})
        if link:
            links.append(link)
    return links


def get_last_assistant_and_prev_user(
    history: Optional[List[Message]]
) -> tuple[Optional[str], Optional[str]]:
    if not history:
        return None, None
    last_assistant = None
    prev_user = None
    last_prev_user = None
    for msg in history:
        if msg.role == "user":
            prev_user = msg.content
        elif msg.role == "assistant":
            last_assistant = msg.content
            last_prev_user = prev_user
    return last_assistant, last_prev_user


async def resolve_selection_from_history(
    user_text: str,
    history: Optional[List[Message]],
    products: List[Dict[str, Any]]
) -> Optional[tuple[List[Dict[str, Any]], Optional[str]]]:
    last_assistant, prev_user = get_last_assistant_and_prev_user(history)
    if not last_assistant:
        return None

    candidates = extract_variant_candidates_from_text(last_assistant, products)
    if not candidates:
        return None

    options_for_llm = []
    for entry in candidates:
        product = entry.get("product") or {}
        variant = entry.get("variant") or {}
        options_for_llm.append({
            "name": product.get("name", "Unknown"),
            "label": variant.get("label") or "Default",
            "price": variant.get("price", 0)
        })

    selection_indices = parse_multi_indices(user_text, len(candidates))
    if not selection_indices:
        selection_indices = await resolve_selection_indexes_with_llm(
            user_text,
            last_assistant,
            options_for_llm
        )

    if not selection_indices:
        return None

    chosen_entries = [candidates[idx] for idx in selection_indices if 0 <= idx < len(candidates)]
    if not chosen_entries:
        return None
    return chosen_entries, prev_user

def pick_cart_candidates(
    user_text: str,
    product_links: List[ProductLink],
    products: List[Dict[str, Any]]
) -> tuple[List[ProductLink], Optional[str]]:
    if not product_links:
        return [], None
    if len(product_links) == 1:
        return product_links, "single"

    text = (user_text or "").strip()
    if not text:
        return [], None

    selection_indices = parse_multi_indices(text, len(product_links))
    if selection_indices:
        return [product_links[i] for i in selection_indices], "index"

    explicit_matches: List[ProductLink] = []
    for link in product_links:
        if user_specified_variant(text, link.variantLabel or "", str(link.variantKey or "")):
            explicit_matches.append(link)

    if explicit_matches:
        if len(explicit_matches) == 1:
            return explicit_matches, "explicit"
        return [], None

    name_matches: List[ProductLink] = []
    lower_text = text.lower()
    for link in product_links:
        if link.name and link.name.lower() in lower_text:
            name_matches.append(link)

    if name_matches:
        if len(name_matches) == 1:
            return name_matches, "name"
        return [], None

    return [], None


def build_cart_actions(
    user_text: str,
    product_links: List[ProductLink],
    products: List[Dict[str, Any]]
) -> List[CartAction]:
    if not should_add_to_cart(user_text):
        return []
    candidates, reason = pick_cart_candidates(user_text, product_links, products)
    if not candidates:
        return []
    qty = parse_quantity(user_text)
    if len(candidates) >= 2 and wants_multiple_distinct_items(user_text):
        qty = 1
    actions: List[CartAction] = []
    for candidate in candidates:
        pid = int(candidate.id or 0)
        cid = int(candidate.combinationId or 0)
        label = candidate.variantLabel or ""
        allow_add = True
        if reason == "name":
            allow_add = product_has_single_variant(products, pid) or user_specified_variant(
                user_text, label, str(candidate.variantKey or "")
            )
        if not allow_add:
            continue
        actions.append(
            CartAction(
                type="cart.add",
                productId=pid,
                combinationId=cid,
                quantity=qty,
                sku=candidate.sku,
                options=candidate.selectedOptions or []
            )
        )
    return actions


def build_cart_actions_from_links(
    product_links: List[ProductLink],
    quantity: int
) -> List[CartAction]:
    actions: List[CartAction] = []
    for link in product_links:
        actions.append(
            CartAction(
                type="cart.add",
                productId=int(link.id or 0),
                combinationId=int(link.combinationId or 0),
                quantity=quantity,
                sku=link.sku,
                options=link.selectedOptions or []
            )
        )
    return actions


def build_cart_confirmation_message(cart_actions: List[CartAction]) -> Optional[str]:
    if not cart_actions:
        return None
    total_qty = sum(action.quantity for action in cart_actions if action.quantity) or len(cart_actions)
    if total_qty == 1:
        return "Added to your cart."
    return f"Added {total_qty} items to your cart."


def merge_product_links(existing: List[ProductLink], new_links: List[ProductLink]) -> List[ProductLink]:
    if not new_links:
        return existing
    combined = list(existing) if existing else []
    existing_keys = {str(p.variantKey).lower() for p in combined if p.variantKey}
    for link in new_links:
        key = str(link.variantKey).lower() if link.variantKey else ""
        if key and key not in existing_keys:
            existing_keys.add(key)
            combined.append(link)
    return combined


def should_include_products(
    user_text: str,
    response_text: str,
    product_links: List[ProductLink],
    pending: Optional[PendingChoice],
    cart_actions: List[CartAction]
) -> bool:
    if cart_actions:
        return False
    if pending and pending.options:
        return True
    if not product_links:
        return False
    if looks_like_recommendation(response_text):
        return True
    if re.search(r"\bwhich one\b|\bchoose\b|\b1 or 2\b", response_text or "", flags=re.IGNORECASE):
        return True
    return False


def resolve_pending_choice(
    user_text: str,
    pending: Optional[PendingChoice],
    products: List[Dict[str, Any]]
) -> tuple[List[CartAction], Optional[PendingChoice]]:
    if not pending or not pending.options:
        return [], None
    selected_links, _ = pick_cart_candidates(user_text, pending.options, products)
    if selected_links:
        qty = pending.quantity or 1
        if len(selected_links) >= 2 and wants_multiple_distinct_items(user_text):
            qty = 1
        return build_cart_actions_from_links(selected_links, qty), None
    text = (user_text or "").strip().lower()
    mentions_option = False
    for option in pending.options:
        if option.name and option.name.lower() in text:
            mentions_option = True
            break
        if user_specified_variant(text, option.variantLabel or "", str(option.variantKey or "")):
            mentions_option = True
            break
    if looks_like_selection_reply(user_text) or mentions_option:
        return [], pending
    return [], None

def build_product_links_from_variant_keys(
    variant_keys: List[str],
    products: List[Dict[str, Any]],
    allowed_variant_keys: Set[str]
) -> List[ProductLink]:
    """Build product link payloads from variant keys and catalog."""
    if not variant_keys or not products:
        return []

    index: Dict[str, Dict[str, Any]] = {}
    for p in products:
        for v in (p.get("variants") or []):
            if isinstance(v, dict) and v.get("in_stock") is False:
                continue
            variant_key = str(v.get("variantKey") or "").strip()
            if variant_key:
                index[variant_key.lower()] = {"product": p, "variant": v}

    links: List[ProductLink] = []
    allowed = {s.strip().lower() for s in allowed_variant_keys if s.strip()}

    for variant_key in variant_keys:
        variant_key_norm = str(variant_key or "").strip().lower()
        if allowed and variant_key_norm not in allowed:
            continue
        entry = index.get(variant_key_norm)
        if not entry:
            continue
        product = entry["product"]
        variant = entry["variant"]
        link = build_product_link(product, variant)
        if link:
            links.append(link)

    return links


def build_product_link(product: Dict[str, Any], variant: Dict[str, Any]) -> Optional[ProductLink]:
    key_value = str(variant.get("variantKey") or "").strip()
    if not key_value:
        return None

    pid = product.get("id")
    cid = variant.get("combinationId")
    parsed_pid, parsed_cid = parse_variant_key_ids(key_value)

    if (pid is None or pid == 0) and parsed_pid is not None:
        pid = parsed_pid
    if cid is None and parsed_cid is not None:
        cid = parsed_cid

    product_options = normalize_product_options(product.get("options"))
    selected_options = normalize_selected_options(variant.get("options"))

    return ProductLink(
        id=int(pid or 0),
        name=product.get("name", "Unknown"),
        combinationId=int(cid or 0),
        variantKey=key_value,
        variantLabel=variant.get("label") or "Default",
        price=float(variant.get("price", 0) or 0),
        sku=(variant.get("effectiveSku") or "").strip() or None,
        url=product.get("url") or None,
        options=product_options,
        selectedOptions=selected_options
    )


def product_has_single_variant(products: List[Dict[str, Any]], product_id: int) -> bool:
    """Return True if a product has exactly one in-stock variant."""
    for product in products:
        try:
            pid = int(product.get("id") or 0)
        except (TypeError, ValueError):
            continue
        if pid != product_id:
            continue
        variants = [
            v for v in (product.get("variants") or [])
            if v.get("variantKey") and v.get("in_stock") is not False
        ]
        return len(variants) == 1
    return False


def _sse_event(payload: Dict[str, Any]) -> str:
    """Format a JSON payload as an SSE data event."""
    return f"data: {json.dumps(payload)}\n\n"

def get_safe_fallback(catalog: Dict[str, Any]) -> tuple[str, List[Dict[str, Any]]]:
    """
    Generate a safe fallback response when model validation fails.
    Returns deterministic recommendations from catalog.
    """
    products = catalog.get("products", [])

    if not products:
        return (
            "I apologize, but I'm having trouble accessing our inventory right now. Please try again in a moment.",
            []
        )

    variants: List[Dict[str, Any]] = []
    for p in products:
        for v in (p.get("variants") or []):
            if not isinstance(v, dict):
                continue
            if v.get("in_stock") is False:
                continue
            if v.get("variantKey") is None:
                continue
            variants.append({"product": p, "variant": v})

    if not variants:
        return (
            "I apologize, but I'm having trouble accessing our inventory right now. Please try again in a moment.",
            []
        )

    # Get top 3 variants (deterministic, no hallucination possible)
    top_variants = variants[:3]

    lines = ["I'm having trouble matching items to our live inventory. Here are some popular options:\n"]

    for entry in top_variants:
        product = entry["product"]
        variant = entry["variant"]
        name = product.get("name", "Unknown")
        label = variant.get("label") or "Default"
        price = variant.get("price", 0)
        lines.append(f"• {name} — {label} - ${price:.2f}")

    lines.append("\nWhat type of product are you looking for? I can help you find exactly what you need.")

    return "\n".join(lines), top_variants
